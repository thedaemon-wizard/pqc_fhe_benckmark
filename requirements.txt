#
# ============ OPTION A: Automatic Installation (Recommended) ================
# liboqs-python can automatically download and build liboqs at runtime
# if it is not detected on the system.
#
#   $ git clone --depth=1 https://github.com/open-quantum-safe/liboqs-python
#   $ cd liboqs-python
#   $ pip install .
#
# ============ OPTION B: Manual Installation =================================
# If you prefer to install liboqs manually first:
#
#   Step 1: Install build dependencies (Ubuntu/Debian)
#   $ sudo apt install cmake gcc libssl-dev python3-pytest
#
#   Step 2: Build and install liboqs C library
#   $ git clone --depth=1 https://github.com/open-quantum-safe/liboqs
#   $ cmake -S liboqs -B liboqs/build -DBUILD_SHARED_LIBS=ON
#   $ cmake --build liboqs/build --parallel 8
#   $ sudo cmake --build liboqs/build --target install
#
#   Step 3: Set library path (Linux)
#   $ export LD_LIBRARY_PATH=$LD_LIBRARY_PATH:/usr/local/lib
#
#   Step 4: Install liboqs-python
#   $ git clone --depth=1 https://github.com/open-quantum-safe/liboqs-python
#   $ cd liboqs-python
#   $ pip install .
